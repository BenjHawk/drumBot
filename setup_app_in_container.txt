source:
https://scotch.io/tutorials/create-a-mean-app-with-angular-2-and-docker-compose

Dockerfile:
FROM alpine
RUN apk aupdate
RUN apk add nodejs
RUN apk add npm
RUN npm install -g @angular/cli
COPY . /app
EXPOSE 4200
WORKDIR /app
CMD ["ng", "serve"]

package.json:
[...]
"start": "ng serve -H 0.0.0.0",
[...]

container mit app ausführen:
docker run -d -p 4200:4200 <image-name>
